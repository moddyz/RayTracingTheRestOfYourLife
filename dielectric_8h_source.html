52  // or traveling within the surface towards the outside.

54  gm::Vec3f incidentNormal;
55  if ( gm::DotProduct( i_ray.Direction(), i_hitRecord.m_normal ) < 0 )
56  {
57  // The incident ray and the normal are opposing, thus the incident ray is outside and
58  // heading into the geometric surface.
59  incidentIndex = c_airRefractiveIndex;
60  refractedIndex = m_refractiveIndex;
61  incidentNormal = i_hitRecord.m_normal;
62  }
63  else
64  {
65  // The incident ray and the normal form an acute angle, thus the incident ray is within the geometric
66  // surface and heading outwards.
67  incidentIndex = m_refractiveIndex;
68  refractedIndex = c_airRefractiveIndex;
69  incidentNormal = -i_hitRecord.m_normal;
70  }
71 
72  gm::Vec3f normRayDir = gm::Normalize( i_ray.Direction() );
73 
74  // Check for total internal reflection (when the ray is inside a material with
75  // higher frefractive index.
76  double cosTheta = gm::Min( gm::DotProduct( -normRayDir, incidentNormal ), 1.0f );
77  double sinTheta = sqrt( 1.0 - cosTheta * cosTheta );
78  if ( ( incidentIndex / refractedIndex ) * sinTheta > 1.0 )
79  {
80  o_scatteredRay = raytrace::Ray( /* origin */ i_hitRecord.m_position,
81  /* direction */ Reflect( normRayDir, incidentNormal ),
82  /* time */ i_ray.Time() );

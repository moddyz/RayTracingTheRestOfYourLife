106  inline void
107  _ComputeNormalAndUV( const gm::Vec3f& i_hitPosition, float i_time, gm::Vec3f& o_normal, gm::Vec2f& o_uv ) const
108  {
109  gm::Vec3f origin = m_origin.Value( i_time );
110  gm::Vec3f dimensions = m_dimensions.Value( i_time );
111  gm::Vec3f originToHit = i_hitPosition - origin;
112 
113  // Divide originToHit by dimensions to remove length bias.
114  originToHit.X() /= ( dimensions.X() * 0.5f );
115  originToHit.Y() /= ( dimensions.Y() * 0.5f );
116  originToHit.Z() /= ( dimensions.Z() * 0.5f );
117 
118  // Choose longest axis as normal.
119  gm::Vec3f originToHitAbs = gm::Abs( originToHit );
120  if ( originToHitAbs.X() > originToHitAbs.Y() && originToHitAbs.X() > originToHitAbs.Z() )
121  {
122  o_normal = gm::Normalize( gm::Vec3f( originToHit.X(), 0, 0 ) );
123  o_uv = gm::Vec2f( origin.Y(), origin.Z() );
124  }
125  else if ( originToHitAbs.Y() > originToHitAbs.X() && originToHitAbs.Y() > originToHitAbs.Z() )
126  {
127  o_normal = gm::Normalize( gm::Vec3f( 0, originToHit.Y(), 0 ) );
128  o_uv = gm::Vec2f( origin.X(), origin.Z() );
129  }
130  else
131  {
132  o_normal = gm::Normalize( gm::Vec3f( 0, 0, originToHit.Z() ) );
133  o_uv = gm::Vec2f( origin.X(), origin.Y() );
134  }
135 
136  // Flip UV coordinates into (0, 1) region if required.
137  if ( o_uv.X() < 0 )
138  {
139  o_uv.X() = 1.0f - o_uv.X();
140  }

143  {
144  o_uv.Y() = 1.0f - o_uv.Y();
145  }

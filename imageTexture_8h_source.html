72  imageCoord.Y() = gm::Clamp( imageCoord.Y(), gm::IntRange( 0, m_imageDim.Y() - 1 ) );

75  constexpr float colorScale = 1.0f / 255.0f;
76 
77  // Get the starting address of the pixel, by offsetting the imageData pointer.
